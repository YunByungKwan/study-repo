interface Language {
    fun translate()
    fun getText(): String
}

class English(private val textDelivery: TextDelivery): Language {
    override fun translate() {
        textDelivery.setText("[English]${textDelivery.getText()}")
    }
    override fun getText(): String = textDelivery.getText()
}

class Spanish(private val textDelivery: TextDelivery): Language {
    override fun translate() {
        textDelivery.setText("[Spanish]${textDelivery.getText()}")
    }
    override fun getText(): String = textDelivery.getText()
}
